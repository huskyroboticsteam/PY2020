17class Factor : public AbstractFactor {
18public:
19 const covariance<D> _sigma_inv;
20 const measurement<D> _measurement;
21
22 Factor(const covariance<D> &sigma_inv, const measurement<D> &measurement) :
23 _sigma_inv(sigma_inv), _measurement(measurement) {}
24
25 virtual measurement<D> f(const values &/*x*/) = 0;
26 virtual jacobian<D> jf(const values &/*x*/) = 0;
27
28 virtual double eval(const values &x) {
29 measurement<D> diff = f(x) - _measurement;
30 return 0.5 * (diff.transpose() * _sigma_inv * diff)(0,0);
31 }
32
33 virtual values gradient_at(const values &x) {
34 return jf(x) * (_sigma_inv * (f(x) - _measurement));
35 }
36
37 virtual hessian hessian_at(const values &x) {
38 jacobian<D> j = jf(x);
39 return j * _sigma_inv * j.transpose();
40 }
41};
