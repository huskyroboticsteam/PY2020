105 eventid_t scheduleEvent(std::chrono::milliseconds period,
106 const std::function<void()>& fn) {
107 eventid_t id;
108 {
109 std::unique_lock lock(scheduleMutex);
110 id = nextID++;
111 auto now = Clock::now();
112 schedule.push({id, now + period, period, fn});
113 newEventAdded = true;

129 void threadFn() {
130 if (name.has_value()) {
131 loguru::set_thread_name(name->c_str());
132 }
133 std::unique_lock lock(scheduleMutex);
134 while (!quitting) {
135 if (schedule.empty()) {
136 scheduleCV.wait(lock, [&] { return newEventAdded; });
137 newEventAdded = false;
138 } else {
139 auto now = Clock::now();
140 schedule_t event = schedule.top();
141 if (toRemove.find(event.id) != toRemove.end()) {
142 // if we should remove this id then remove the event and don't reschedule
143 schedule.pop();
144 toRemove.erase(event.id);
145 } else if (event.nextSendTime <= now) {
146 // execute the event and reschedule it
147 schedule.pop();
148 event.fn();
149 event.nextSendTime += event.period;
150 schedule.push(event);
151 } else {
152 // wait until the event should be executed
153 scheduleCV.wait_until(lock, event.nextSendTime,
154 [&] { return newEventAdded; });
155 newEventAdded = false;
156 }
157 }
158 }
