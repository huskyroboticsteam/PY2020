52 void set_setpoint(const navtypes::Vectord<N>& targetJointPos) {
53 Eigen::Vector2d newSetPoint = kin.jointPosToEEPos(targetJointPos);
54 std::lock_guard<std::mutex> lock(mutex);
55 setpoint = normalizeEEWithinRadius(newSetPoint);

64 Eigen::Vector2d get_setpoint(robot::types::datatime_t currTime) {
65 Eigen::Vector2d pos;
66 {
67 std::lock_guard<std::mutex> lock(mutex);
68 // calculate current EE setpoint
69 pos = setpoint;
70 if (velTimestamp.has_value()) {
71 double dt = util::durationToSec(currTime - velTimestamp.value());
72 pos += velocity * dt;
73 }
74 }
75
76 // bounds check (new pos + vel vector <= sum of joint lengths)
77 return normalizeEEWithinRadius(pos);
78 }

124 navtypes::Vectord<N> getCommand(robot::types::datatime_t currTime,
125 const navtypes::Vectord<N>& currJointPos) {
126 Eigen::Vector2d newPos = get_setpoint(currTime);
127 // lock after calling get_setpoint since that internally locks the mutex
128 std::lock_guard<std::mutex> lock(mutex);
129 setpoint = newPos;
130
131 // get new joint positions for target EE
132 return kin.eePosToJointPos(newPos, currJointPos);
133 }

149 Eigen::Vector2d normalizeEEWithinRadius(Eigen::Vector2d eePos) {
150 double radius = kin.getSegLens().sum() * safetyFactor;
151
152 if (eePos.norm() > radius) {
153 // TODO: will need to eventually shrink velocity vector until it is within radius
154 // instead of just normalizing it.
155
156 eePos.normalize();
157 eePos *= radius;
158 }
159
160 return eePos;
161 }
