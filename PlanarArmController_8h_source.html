54 Eigen::Vector2d get_setpoint(robot::types::datatime_t currTime) {
55 Eigen::Vector2d pos;
56 {
57 std::lock_guard<std::mutex> lock(mutex);
58 // calculate current EE setpoint
59 pos = setpoint;
60 if (velTimestamp.has_value()) {
61 double dt = util::durationToSec(currTime - velTimestamp.value());
62 pos += velocity * dt;
63 }
64 }
65
66 // bounds check (new pos + vel vector <= sum of joint lengths)
67 double radius = kin.getSegLens().sum();
68 if (pos.norm() > radius) {
69 // new position is outside of bounds
70 // TODO: will need to eventually shrink velocity vector until it is within radius
71 // instead of just normalizing it
72 pos.normalize();
73 }
74 return pos;
75 }
