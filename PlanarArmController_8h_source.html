93 static bool is_setpoint_valid(const navtypes::Vectord<N>& targetJointPos,
94 kinematics::PlanarArmKinematics<N> kin_obj,
95 double safetyFactor) {
96 // Compute the new EE position to determine if it is within
97 // safety factor * length of fully extended arm.
98 double eeRadius = kin_obj.jointPosToEEPos(targetJointPos).norm();
99 double maxRadius = kin_obj.getSegLens().sum() * safetyFactor;

120 Eigen::Vector2d get_setpoint(robot::types::datatime_t currTime) {
121 Eigen::Vector2d pos;
122 {
123 std::lock_guard<std::mutex> lock(mutex);
124 // calculate current EE setpoint
125 pos = mutableFields->setpoint;
126 if (mutableFields->velTimestamp.has_value()) {
127 double dt =
128 util::durationToSec(currTime - mutableFields->velTimestamp.value());
129 pos += mutableFields->velocity * dt;
130 }
131 }
132
133 // bounds check (new pos + vel vector <= sum of joint lengths)
134 return normalizeEEWithinRadius(pos);

185 navtypes::Vectord<N> getCommand(robot::types::datatime_t currTime,
186 const navtypes::Vectord<N>& currJointPos) {
187 Eigen::Vector2d newPos = get_setpoint(currTime);
188 // lock after calling get_setpoint since that internally locks the mutex
189 std::lock_guard<std::mutex> lock(mutex);
190
191 // get new joint positions for target EE
192 bool success = false;
193 navtypes::Vectord<N> jp = kin.eePosToJointPos(newPos, currJointPos, success);

198 } else {

222 Eigen::Vector2d normalizeEEWithinRadius(Eigen::Vector2d eePos) {
223 double radius = kin.getSegLens().sum() * safetyFactor;
224 if (eePos.norm() > (radius + 1e-4)) {
225 // new position is outside of bounds. Set new EE setpoint so it will follow the
226 // velocity vector instead of drifting along the radius.
227 // setpoint = setpoint inside circle
228 // eePos = new setpoint outside circle
229 // radius = ||setpoint + a*(eePos - setpoint)|| solve for a
230 double diffDotProd =
231 (eePos - mutableFields->setpoint).dot(mutableFields->setpoint);
232 double differenceNorm = (eePos - mutableFields->setpoint).squaredNorm();
233 double discriminant =
234 std::pow(diffDotProd, 2) -
235 differenceNorm * (mutableFields->setpoint.squaredNorm() - std::pow(radius, 2));
236 double a = (-diffDotProd + std::sqrt(discriminant)) / differenceNorm;
237 // new constrained eePos = (1 - a) * (ee inside circle) + a * (ee outside circle)
238 eePos = (1 - a) * mutableFields->setpoint + a * eePos;
239 }
240 return eePos;
241 }
