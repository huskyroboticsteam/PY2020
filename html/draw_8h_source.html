18 void Drawellipse(const Eigen::VectorXd& X, float a, float b, string color) {
19 
20  assert(X.size() == 3);
21  const int NPOINTS = 100;
22  Eigen::VectorXd vec = Eigen::VectorXd::LinSpaced(NPOINTS, 0, 2 * M_PI);
23  int N = vec.size();
24 
25  Eigen::MatrixXd p = Eigen::MatrixXd::Ones(2,N);
26  p.row(0) = a * vec.array().cos();
27  p.row(1) = b * vec.array().sin();
28 
29  // //handle the rotation and translation
30  float x0 = X(0), y0 = X(1), angle = X(2);
31  Eigen::MatrixXd R = Eigen::MatrixXd(2,2);
32  R << cos(angle), -sin(angle), sin(angle), cos(angle);
33 
34  Eigen::MatrixXd T = Eigen::MatrixXd::Ones(2,N);
35  T.row(0) = x0*T.row(0);
36  T.row(1) = y0*T.row(1);
37  p = R*p + T;
38 
39  //for plotting purpuse only
40  vector<float> Px, Py;
41  for (int i = 0; i < N; i++) {
42  Px.push_back(p(0,i));
43  Py.push_back(p(1,i));
44  }
45  plt::plot(Px, Py, color);
46  }

48  void Drawprobellipse(VectorXd x, const MatrixXd& C, float alpha, string color) {
49 
50  float sxx = C(0, 0), syy = C(1, 1), sxy = C(0, 1);
51  float a = sqrt(0.5*(sxx+syy+sqrt(pow(sxx-syy, 2)+4*pow(sxy,2)))); // always greater
52  float b = sqrt(0.5*(sxx+syy-sqrt(pow(sxx-syy, 2)+4*pow(sxy,2)))); // always smaller
53 
54  //% Scaling in order to reflect specified probability
55  a = a*sqrt(chi2invtable(alpha,2));
56  b = b*sqrt(chi2invtable(alpha,2));
57 
58  if (sxx < syy) {
59  float temp = a;
60  a = b;
61  b = temp;
62  }
63 
64  float angle = 0;
65  //% Calculate inclination (numerically stable)
66  if (sxx != syy)
67  angle = 0.5*atan(2*sxy/(sxx-syy));
68  else if (sxy == 0)
69  angle = 0; //angle doesn't matter
70  else if (sxy > 0)
71  angle = M_PI/4;
72  else if (sxy < 0)
73  angle = -M_PI/4;
74  x(2) = angle;
75  Drawellipse(x,a,b,color);
76  }
