55 navtypes::Matrixd<2, N> getJacobian(const navtypes::Vectord<N>& jointPos) const {
56 Eigen::Matrix<double, 2, N> jacobian = Eigen::Matrix<double, 2, N>::Zero();
57 std::array<double, N> thetaCumSum;
58 thetaCumSum[0] = jointPos[0];
59 for (unsigned int i = 1; i < N; i++) {
60 thetaCumSum[i] = jointPos[i] + thetaCumSum[i - 1];
61 }
62 for (unsigned int i = 0; i < N; i++) {
63 for (unsigned int j = i; j < N; j++) {
64 jacobian(0, i) += -segLens[j] * std::sin(thetaCumSum[j]);
65 jacobian(1, i) += segLens[j] * std::cos(thetaCumSum[j]);
66 }
67 }
68 return jacobian;
69 }
