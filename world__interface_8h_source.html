248template <unsigned long int N>
249types::DataPoint<navtypes::Vectord<N>>
250getMotorPositionsRad(const std::array<types::motorid_t, N>& motors) {
251 navtypes::Vectord<N> motorPositions;
252 std::optional<types::datatime_t> timestamp;
253 for (size_t i = 0; i < motors.size(); i++) {
254 types::DataPoint<int32_t> positionDataPoint = robot::getMotorPos(motors[i]);
255 if (positionDataPoint.isValid()) {
256 if (!timestamp.has_value() || timestamp > positionDataPoint.getTime()) {
257 timestamp = positionDataPoint.getTime();
258 }
259 double position = static_cast<double>(positionDataPoint.getData());
260 position *= M_PI / 180.0 / 1000.0;
261 motorPositions(i) = position;
262 } else {
263 return {};
264 }
265 }
266 return types::DataPoint<navtypes::Vectord<N>>(timestamp.value(), motorPositions);
267}
