243template <unsigned long int N>
244types::DataPoint<navtypes::Vectord<N>>
245getMotorPositionsRad(const std::array<types::motorid_t, N>& motors) {
246 navtypes::Vectord<N> motorPositions;
247 std::optional<types::datatime_t> timestamp;
248 for (size_t i = 0; i < motors.size(); i++) {
249 types::DataPoint<int32_t> positionDataPoint = robot::getMotorPos(motors[i]);
250 if (positionDataPoint.isValid()) {
251 if (!timestamp.has_value() || timestamp > positionDataPoint.getTime()) {
252 timestamp = positionDataPoint.getTime();
253 }
254 double position = static_cast<double>(positionDataPoint.getData());
255 position *= M_PI / 180.0 / 1000.0;
256 motorPositions(i) = position;
257 } else {
258 return {};
259 }
260 }
261 return types::DataPoint<navtypes::Vectord<N>>(timestamp.value(), motorPositions);
262}
