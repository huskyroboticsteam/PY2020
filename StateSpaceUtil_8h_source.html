79 template <int numStates, int numInputs>
80 void continuousToDiscrete(Eigen::Matrix<double, numStates, numStates> &A,
81  Eigen::Matrix<double, numStates, numInputs> &B, double dt)
82 {
83  // zero order hold discretization for system and input matrices
84  // reference:
85  // https://en.wikipedia.org/wiki/Discretization#Discretization_of_linear_state_space_models
86  Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> M;
87  M.setZero();
88  M.template block<numStates, numStates>(0, 0) = A;
89  M.template block<numStates, numInputs>(0, numStates) = B;
90  M *= dt;
91 
92  Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> exp = M.exp();
93  A = exp.template block<numStates, numStates>(0, 0);
94  B = exp.template block<numStates, numInputs>(0, numStates);
95 }

124 template <int numStates>
125 void discretizeAQ(const Eigen::Matrix<double, numStates, numStates> &contA,
126  const Eigen::Matrix<double, numStates, numStates> &contQ,
127  Eigen::Matrix<double, numStates, numStates> &discA,
128  Eigen::Matrix<double, numStates, numStates> &discQ, double dt)
129 {
130  // zero order hold discretization of the system and state covariance matrices
131  // reference:
132  // https://en.wikipedia.org/wiki/Discretization#Discretization_of_process_noise
133  Eigen::Matrix<double, numStates * 2, numStates * 2> M;
134  M.template block<numStates, numStates>(0, 0) = -contA;
135  M.template block<numStates, numStates>(0, numStates) = contQ;
136  M.template block<numStates, numStates>(numStates, 0).setZero();
137  M.template block<numStates, numStates>(numStates, numStates) = contA.transpose();
138  M *= dt;
139 
140  Eigen::Matrix<double, numStates * 2, numStates * 2> G = M.exp();
141 
142  Eigen::Matrix<double, numStates, numStates> AinvQ =
143  G.block(0, numStates, numStates, numStates); // this is discA^-1 * Q
144  discA = G.block(numStates, numStates, numStates, numStates).transpose();
145 
146  Eigen::Matrix<double, numStates, numStates> q = discA * AinvQ; // A * A^-1 * Q = Q
147  discQ = (q + q.transpose()) / 2.0; // make Q symmetric again if it became asymmetric
148 }

186 // matrix This can be used to calculate the optimal Kalman gain matrix
199 template <int numStates>
200 static Eigen::Matrix<double, numStates, numStates>
201 DARE(const Eigen::Matrix<double, numStates, numStates> &A0,
202  const Eigen::Matrix<double, numStates, numStates> &Ctrans,
203  const Eigen::Matrix<double, numStates, numStates> &Q,
204  const Eigen::Matrix<double, numStates, numStates> &R)
205 {
206  // reference:
207  // https://scicomp.stackexchange.com/questions/30757/discrete-time-algebraic-riccati-equation-dare-solver-in-c
208  Eigen::Matrix<double, numStates, numStates> A = A0;
209  Eigen::Matrix<double, numStates, numStates> G = Ctrans * R.inverse() * Ctrans.transpose();
210  Eigen::Matrix<double, numStates, numStates> H = Q;
211  const Eigen::Matrix<double, numStates, numStates> I =
212  Eigen::Matrix<double, numStates, numStates>::Identity();
213 
214  Eigen::Matrix<double, numStates, numStates> lastA;
215  Eigen::Matrix<double, numStates, numStates> lastG;
216  Eigen::Matrix<double, numStates, numStates> lastH;
217  // converges quadratically
218  do
219  {
220  lastA = A;
221  lastG = G;
222  lastH = H;
223 
224  const Eigen::Matrix<double, numStates, numStates> AIGH =
225  lastA * (I + lastG * lastH).inverse();
226  A = lastA * AIGH * lastA;
227  G = lastG + lastA * AIGH * lastG * lastA.transpose();
228  H = lastH + lastA.transpose() * lastH * (I + lastG * lastH).transpose() * lastA;
229  } while ((H - lastH).norm() / H.norm() >= 1e-4);
230 
231  return H;
232 }
