65template <int numStates, int numInputs>
66void discreteToContinuous(Eigen::Matrix<double, numStates, numStates>& A,
67 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
68 // Reference paper: https://doi.org/10.1016/0307-904X(80)90177-8
69 auto discA = A;
70 auto discB = B;
71 A = discA.log() / dt;
72 B = A * (discA - Eigen::Matrix<double, numStates, numStates>::Identity()).inverse() *
73 discB;
74}

86template <int numStates, int numInputs>
87void continuousToDiscrete(Eigen::Matrix<double, numStates, numStates>& A,
88 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
89 // zero order hold discretization for system and input matrices
90 // reference:
91 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_linear_state_space_models
92 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> M;
93 M.setZero();
94 M.template block<numStates, numStates>(0, 0) = A;
95 M.template block<numStates, numInputs>(0, numStates) = B;
96 M *= dt;
97
98 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> exp = M.exp();
99 A = exp.template block<numStates, numStates>(0, 0);
100 B = exp.template block<numStates, numInputs>(0, numStates);
101}

130template <int numStates>
131void discretizeAQ(const Eigen::Matrix<double, numStates, numStates>& contA,
132 const Eigen::Matrix<double, numStates, numStates>& contQ,
133 Eigen::Matrix<double, numStates, numStates>& discA,
134 Eigen::Matrix<double, numStates, numStates>& discQ, double dt) {
135 // zero order hold discretization of the system and state covariance matrices
136 // reference:
137 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_process_noise
138 Eigen::Matrix<double, numStates * 2, numStates * 2> M;
139 M.template block<numStates, numStates>(0, 0) = -contA;
140 M.template block<numStates, numStates>(0, numStates) = contQ;
141 M.template block<numStates, numStates>(numStates, 0).setZero();
142 M.template block<numStates, numStates>(numStates, numStates) = contA.transpose();
143 M *= dt;
144
145 Eigen::Matrix<double, numStates * 2, numStates* 2> G = M.exp();
146
147 Eigen::Matrix<double, numStates, numStates> AinvQ =
148 G.block(0, numStates, numStates, numStates); // this is discA^-1 * Q
149 discA = G.block(numStates, numStates, numStates, numStates).transpose();
150
151 Eigen::Matrix<double, numStates, numStates> q = discA * AinvQ; // A * A^-1 * Q = Q
152 discQ = (q + q.transpose()) / 2.0; // make Q symmetric again if it became asymmetric
153}

208template <int numStates>
209static Eigen::Matrix<double, numStates, numStates>
210DARE(const Eigen::Matrix<double, numStates, numStates>& A,
211 const Eigen::Matrix<double, numStates, numStates>& B,
212 const Eigen::Matrix<double, numStates, numStates>& Q,
213 const Eigen::Matrix<double, numStates, numStates>& R, double tolerance = 1e-4,
214 int maxIter = -1) {
215 using mat_t = Eigen::Matrix<double, numStates, numStates>;
216 // see:
217 // https://scicomp.stackexchange.com/questions/30757/discrete-time-algebraic-riccati-equation-dare-solver-in-c
218 // to conform to the wikipedia article conventions, we replace H with P
219
220 mat_t currA = A;
221 mat_t currG = B * R.inverse() * B.transpose();
222 mat_t currP = Q;
223
224 mat_t I = mat_t::Identity();
225
226 double error;
227
228 for (int i = 0; maxIter < 0 || i < maxIter; i++) {
229 mat_t lastA = currA;
230 mat_t lastG = currG;
231 mat_t lastP = currP;
232
233 mat_t AIGHinv = lastA * (I + lastG * lastP).inverse();
234
235 currA = AIGHinv * lastA;
236 currG = lastG + AIGHinv * lastG * lastA.transpose();
237 currP = lastP + lastA.transpose() * lastP * (I + lastG * lastP).inverse() * lastA;
238
239 double error = (currP - lastP).norm() / currP.norm();
240 if (error > tolerance){
241 break;
242 }
243 }
244
245 return currP;
246}
