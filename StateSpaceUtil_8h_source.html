54template <int numStates, int numInputs>
55void discreteToContinuous(Eigen::Matrix<double, numStates, numStates>& A,
56 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
57 // Reference paper: https://doi.org/10.1016/0307-904X(80)90177-8
58 auto discA = A;
59 auto discB = B;
60 A = discA.log() / dt;
61 B = A * (discA - Eigen::Matrix<double, numStates, numStates>::Identity()).inverse() *
62 discB;
63}

75template <int numStates, int numInputs>
76void continuousToDiscrete(Eigen::Matrix<double, numStates, numStates>& A,
77 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
78 // zero order hold discretization for system and input matrices
79 // reference:
80 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_linear_state_space_models
81 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> M;
82 M.setZero();
83 M.template block<numStates, numStates>(0, 0) = A;
84 M.template block<numStates, numInputs>(0, numStates) = B;
85 M *= dt;
86
87 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> exp = M.exp();
88 A = exp.template block<numStates, numStates>(0, 0);
89 B = exp.template block<numStates, numInputs>(0, numStates);
90}

118template <int numStates>
119void discretizeAQ(const Eigen::Matrix<double, numStates, numStates>& contA,
120 const Eigen::Matrix<double, numStates, numStates>& contQ,
121 Eigen::Matrix<double, numStates, numStates>& discA,
122 Eigen::Matrix<double, numStates, numStates>& discQ, double dt) {
123 // zero order hold discretization of the system and state covariance matrices
124 // reference:
125 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_process_noise
126 Eigen::Matrix<double, numStates * 2, numStates * 2> M;
127 M.template block<numStates, numStates>(0, 0) = -contA;
128 M.template block<numStates, numStates>(0, numStates) = contQ;
129 M.template block<numStates, numStates>(numStates, 0).setZero();
130 M.template block<numStates, numStates>(numStates, numStates) = contA.transpose();
131 M *= dt;
132
133 Eigen::Matrix<double, numStates * 2, numStates* 2> G = M.exp();
134
135 Eigen::Matrix<double, numStates, numStates> AinvQ =
136 G.block(0, numStates, numStates, numStates); // this is discA^-1 * Q
137 discA = G.block(numStates, numStates, numStates, numStates).transpose();
138
139 Eigen::Matrix<double, numStates, numStates> q = discA * AinvQ; // A * A^-1 * Q = Q
140 discQ = (q + q.transpose()) / 2.0; // make Q symmetric again if it became asymmetric
141}

176// Solves Discrete-time Algebraic Riccati Equation to calculate asymptotic error covariance
177// matrix This can be used to calculate the optimal Kalman gain matrix
190template <int numStates>
191static Eigen::Matrix<double, numStates, numStates>
192DARE(const Eigen::Matrix<double, numStates, numStates>& A0,
193 const Eigen::Matrix<double, numStates, numStates>& Ctrans,
194 const Eigen::Matrix<double, numStates, numStates>& Q,
195 const Eigen::Matrix<double, numStates, numStates>& R) {
196 // reference:
197 // https://scicomp.stackexchange.com/questions/30757/discrete-time-algebraic-riccati-equation-dare-solver-in-c
198 Eigen::Matrix<double, numStates, numStates> A = A0;
199 Eigen::Matrix<double, numStates, numStates> G = Ctrans * R.inverse() * Ctrans.transpose();
200 Eigen::Matrix<double, numStates, numStates> H = Q;
201 const Eigen::Matrix<double, numStates, numStates> I =
202 Eigen::Matrix<double, numStates, numStates>::Identity();
203
204 Eigen::Matrix<double, numStates, numStates> lastA;
205 Eigen::Matrix<double, numStates, numStates> lastG;
206 Eigen::Matrix<double, numStates, numStates> lastH;
207 // converges quadratically
208 do {
209 lastA = A;
210 lastG = G;
211 lastH = H;
212
213 const Eigen::Matrix<double, numStates, numStates> AIGH =
214 lastA * (I + lastG * lastH).inverse();
215 A = lastA * AIGH * lastA;
216 G = lastG + lastA * AIGH * lastG * lastA.transpose();
217 H = lastH + lastA.transpose() * lastH * (I + lastG * lastH).transpose() * lastA;
218 } while ((H - lastH).norm() / H.norm() >= 1e-4);
219
220 return H;
221}
