118 void predict(const Eigen::Matrix<double, inputDim, 1>& input) override {
119 Eigen::Matrix<double, stateDim, stateDim> F =
120 getStateFuncJacobianX(stateFunc, this->xHat, input);
121 Eigen::Matrix<double, processNoiseDim, processNoiseDim> processNoise =
122 Q.get(this->xHat, input);
123
124 Eigen::Matrix<double, stateDim, processNoiseDim> L =
125 getStateFuncJacobianW(stateFunc, this->xHat, input);
126
127 this->xHat = stateFunc(this->xHat, input, processnoise_t::Zero());
128 this->P = F * this->P * F.transpose() + L * processNoise * L.transpose();
129 }
130
131 void correct(const Eigen::Matrix<double, outputDim, 1>& measurement) override {
132 Eigen::Matrix<double, outputDim, stateDim> H =
133 getOutputFuncJacobianX(outputFunc, this->xHat); // output matrix
134
135 Eigen::Matrix<double, outputNoiseDim, outputNoiseDim> outputNoise =
136 R.get(this->xHat, measurement);
137 Eigen::Matrix<double, outputDim, outputNoiseDim> M =
138 getOutputFuncJacobianV(outputFunc, this->xHat);
139 Eigen::Matrix<double, outputDim, outputDim> S =
140 H * this->P * H.transpose() +
141 M * outputNoise * M.transpose(); // residual covariance
142 // near-optimal kalman gain
143 Eigen::Matrix<double, stateDim, outputDim> K =
144 S.transpose().colPivHouseholderQr().solve(H * this->P.transpose()).transpose();
145 output_t y = measurement -
146 outputFunc(this->xHat, outputnoise_t::Zero()); // measurement residual
147
148 this->xHat = this->xHat + K * y;
149 this->P = (Eigen::Matrix<double, stateDim, stateDim>::Identity() - K * H) * this->P;
150 }

191 getStateFuncJacobianX(const statefunc_t& f, const state_t& x, const input_t& u) const {
192 processnoise_t w = processnoise_t::Zero();
193 // If we have an analytic solution, use that
194 if (stateFuncJacobianX) {
195 return stateFuncJacobianX(x, u, w);
196 } else {
197 Eigen::Matrix<double, stateDim, stateDim> jacobian;
198 for (int i = 0; i < stateDim; i++) {
199 state_t delta = state_t::Zero();
200 delta[i] = epsilon;
201 state_t derivative = (f(x + delta, u, w) - f(x - delta, u, w)) / (2 * epsilon);
202 jacobian.col(i) = derivative;
203 }
