120 void predict(const Eigen::Matrix<double, inputDim, 1>& input) override {
121 Eigen::Matrix<double, stateDim, stateDim> F =
122 getStateFuncJacobianX(stateFunc, this->xHat, input);
123 Eigen::Matrix<double, processNoiseDim, processNoiseDim> processNoise =
124 Q.get(this->xHat, input);
125
126 Eigen::Matrix<double, stateDim, processNoiseDim> L =
127 getStateFuncJacobianW(stateFunc, this->xHat, input);
128
129 this->xHat = stateFunc(this->xHat, input, processnoise_t::Zero());
130 this->P = F * this->P * F.transpose() + L * processNoise * L.transpose();
131 }
132
133 void correct(const Eigen::Matrix<double, outputDim, 1>& measurement) override {
134 Eigen::Matrix<double, outputDim, stateDim> H =
135 getOutputFuncJacobianX(outputFunc, this->xHat); // output matrix
136
137 Eigen::Matrix<double, outputNoiseDim, outputNoiseDim> outputNoise =
138 R.get(this->xHat, measurement);
139 Eigen::Matrix<double, outputDim, outputNoiseDim> M =
140 getOutputFuncJacobianV(outputFunc, this->xHat);
141 Eigen::Matrix<double, outputDim, outputDim> S =
142 H * this->P * H.transpose() +
143 M * outputNoise * M.transpose(); // residual covariance
144 // near-optimal kalman gain
145 Eigen::Matrix<double, stateDim, outputDim> K =
146 S.transpose().colPivHouseholderQr().solve(H * this->P.transpose()).transpose();
147 output_t y = measurement -
148 outputFunc(this->xHat, outputnoise_t::Zero()); // measurement residual
149
150 this->xHat = this->xHat + K * y;
151 this->P = (Eigen::Matrix<double, stateDim, stateDim>::Identity() - K * H) * this->P;
152 }
